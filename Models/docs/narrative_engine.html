Psychological Narrative Engine - Multi-NPC Conversation Engine

A Fallout-style psychological narrative engine where:

  • Player choices are visible and structured (node/choice tree)
  • NPC responses are generated via the BDI+LLM pipeline
  • Scenarios are NPC-agnostic; NPC individuality is injected at runtime
  • Multiple NPCs can participate independently in the same scenario
  • Routing between nodes is DYNAMIC: driven by NPC BDI state + player_relation

Author: Jerome Bawa (original), refactor by AI assistant

138    def start_session(
139        self,
140        npc_ids: List[str],
141        scenario_id: str,
142        player_skills: Optional[PlayerSkillSet] = None,
143    ) -> ConversationSession:
144        """
145        Start a new conversation session with one or more NPCs.
146
147        Creates a DialogueProcessor for each NPC, injecting their personality
148        into the NPC-agnostic scenario structure.
149
150        Args:
151            npc_ids: List of NPC IDs to participate in this conversation
152            scenario_id: ID of the scenario to use
153            player_skills: Optional player skill set (defaults to 5 in all skills)
154
155        Returns:
156            ConversationSession object for the active conversation
157
158        Raises:
159            ValueError: If scenario_id or any npc_id is not loaded
160        """
161        if scenario_id not in self.scenarios:
162            raise ValueError(f"Scenario '{scenario_id}' not loaded")
163        for npc_id in npc_ids:
164            if npc_id not in self.npcs:
165                raise ValueError(f"NPC '{npc_id}' not loaded")
166
167        scenario = self.scenarios[scenario_id]
168        npc_intent = ScenarioLoader.parse_npc_intent(scenario.get("npc_intent", {}))
169        npc_role = ScenarioLoader.parse_npc_role(scenario.get("npc_role", {}))
170
171        # attach parsed npc_role onto the scenario object for later use
172        scenario["_npc_role_meta"] = npc_role
173
174        if player_skills is None:
175            player_skills = PlayerSkillSet(authority=5, diplomacy=5, empathy=5, manipulation=5)
176
177        npc_states: Dict[str, NPCConversationState] = {}
178        for npc_id in npc_ids:
179            npc = self.npcs[npc_id]
180
181            start_node = ScenarioLoader.get_node(scenario, "start")
182            if not start_node or not start_node.get("choices"):
183                outcome_index = OutcomeIndex(choice_id="empty", interaction_outcomes=[], terminal_outcomes=[])
184            else:
185                outcome_index = ScenarioLoader.parse_outcome_index(start_node["choices"][0])
186
187            processor = DialogueProcessor(
188                npc=npc,
189                player_skills=player_skills,
190                npc_intent=npc_intent,
191                outcome_index=outcome_index,
192                conversation_id=f"{scenario_id}:{npc_id}",
193                use_ollama=self.use_ollama,
194                ollama_url=self.ollama_url,
195            )
196
197            npc_states[npc_id] = NPCConversationState(
198                npc_id=npc_id,
199                npc=npc,
200                processor=processor,
201                scenario_id=scenario_id,
202                current_node="start",
203            )
204
205        session_id = self._make_session_id(npc_ids, scenario_id)
206        session = ConversationSession(scenario_id=scenario_id, scenario=scenario, npc_states=npc_states)
207        self.sessions[session_id] = session
208
209        self._display_opening(session)
210        return session

307    def apply_choice(
308        self,
309        session: ConversationSession,
310        node_id: str,
311        choice_index: int,
312    ) -> Dict[str, Any]:
313        """
314        Apply a single player choice (by index) to all active NPCs.
315
316        Flow per NPC:
317          1. Parse choice → PlayerDialogueInput
318          2. Run through DialogueProcessor (BDI + LLM)
319          3. Evaluate current node's transitions against updated NPC state
320          4. Route NPC to the resolved target node (or stay if no match)
321
322        This is where the dynamic routing happens: same choice, potentially
323        different outcomes for each NPC based on their internal state.
324
325        Args:
326            session: Current conversation session
327            node_id: ID of the current node
328            choice_index: Index of the player's choice (1-based)
329
330        Returns:
331            Dict mapping npc_id to {context, resolved_node}
332
333        Raises:
334            ValueError: If node_id is invalid or choice_index is out of range
335        """
336        node = self.get_node(session, node_id)
337        if not node:
338            raise ValueError(f"Node '{node_id}' not found in scenario '{session.scenario_id}'")
339
340        choices = node.get("choices", [])
341        if not (1 <= choice_index <= len(choices)):
342            raise ValueError(f"Invalid choice index {choice_index} for node '{node_id}'")
343
344        choice_data = choices[choice_index - 1]
345
346        # Fallout-style visible player choice list
347        print("Available choices:")
348        for idx, c in enumerate(choices, start=1):
349            marker = ">>" if idx == choice_index else "  "
350            print(f"{marker} [{idx}] {c['text']}")
351
352        player_input = ScenarioLoader.parse_player_input(choice_data)
353        session.player_choice_log.append(
354            {
355                "node_id": node_id,
356                "choice_id": choice_data["choice_id"],
357                "text": player_input.choice_text,
358            }
359        )
360
361        responses: Dict[str, Any] = {}
362        outcome_index = ScenarioLoader.parse_outcome_index(choice_data)
363        transitions = node.get("transitions", [])
364
365        for state in session.active_npcs():
366            print(f"\nPlayer → {state.npc.name}: {player_input.choice_text}")
367            state.add_exchange("Player", player_input.choice_text)
368            state.choices_made.append(choice_data["choice_id"])
369
370            # --- Step 1-2: Process dialogue through BDI pipeline ---
371            state.processor.outcome_index = outcome_index
372            context = state.processor.process_dialogue(
373                player_input,
374                generate_with_ollama=self.use_ollama,
375            )
376
377            thought = context["thought_reaction"]
378            desire = context["desire_state"]
379            intention = context["behavioural_intention"]
380            npc_response = context["npc_response"]
381
382            # If npc_response comes from scenario prompts containing parsewords,
383            # run it through the formatter for this NPC.
384            npc_response = self._format_with_npc(npc_response, state.npc.name)
385
386            print(f"\n[{state.npc.name} | Internal Thought]: {thought['internal_thought']}")
387            print(
388                f"[{state.npc.name} | Desire]: "
389                f"{desire['immediate_desire']} "
390                f"(type={desire['desire_type']}, intensity={desire['intensity']:.2f})"
391            )
392            print(f"[{state.npc.name} | Intention]: {intention['intention_type']}")
393            print(f"\n{state.npc.name}: {npc_response}\n")
394
395            state.add_exchange(
396                state.npc.name,
397                npc_response,
398                metadata={
399                    "thought": thought,
400                    "desire": desire,
401                    "intention": intention,
402                    "interaction_outcome": context.get("interaction_outcome"),
403                },
404            )
405
406            # --- Step 3: Resolve transitions against this NPC's live state ---
407            resolved_node = TransitionResolver.resolve(transitions, state)
408
409            if resolved_node:
410                target_node = self.get_node(session, resolved_node)
411
412                # --- Step 4a: Terminal node reached ---
413                if target_node and TransitionResolver.is_terminal(target_node):
414                    terminal_id = target_node.get("terminal_id", "unknown")
415                    terminal_result = target_node.get("terminal_result", "")
416                    # Terminal dialogue is generated by the NPC via its prompt,
417                    # not hardcoded. Use npc_dialogue_prompt as the LLM's guide.
418                    terminal_dialogue = npc_response  # already formatted above
419
420                    state.is_complete = True
421                    state.terminal_outcome = {
422                        "terminal_id": terminal_id,
423                        "result": terminal_result,
424                        "final_dialogue": terminal_dialogue,
425                    }
426                    state.current_node = resolved_node
427
428                    print(f"\n{'=' * 70}")
429                    print(f"TERMINAL OUTCOME for {state.npc.name}: {terminal_id.upper()}")
430                    print(f"Result: {terminal_result}")
431                    print(f"{state.npc.name}: {terminal_dialogue}")
432                    print(f"{'=' * 70}\n")
433
434                    state.processor.end_conversation()
435
436                # --- Step 4b: Non-terminal node; advance normally ---
437                else:
438                    state.current_node = resolved_node
439                    print(f"  [{state.npc.name}] → routed to node: {resolved_node}")
440            else:
441                # No transition matched. Stay on current node if it has choices,
442                # or fall through to 'probing' as a default continuation.
443                default_node = "probing"
444                if self.get_node(session, default_node):
445                    state.current_node = default_node
446                    print(f"  [{state.npc.name}] → no transition matched, defaulting to: {default_node}")
447                else:
448                    print(f"  [{state.npc.name}] → no transition matched, staying on: {node_id}")
449
450            responses[state.npc_id] = {
451                "context": context,
452                "resolved_node": state.current_node,
453            }
454
455        return responses
