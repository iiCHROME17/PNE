 75@dataclass
 76class CognitiveModel:
 77    """Represents NPC's internal psychological state"""
 78    self_esteem: float = 0.5  # 0-1: confidence and self-worth
 79    locus_of_control: float = 0.5  # 0=external (blames fate), 1=internal (self-responsible)
 80    cog_flexibility: float = 0.5  # 0=rigid/dogmatic, 1=adaptive/open
 81    
 82    def __post_init__(self):
 83        self._validate_ranges()
 84    
 85    def _validate_ranges(self):
 86        """Ensure all values are within 0-1 range"""
 87        for attr in ['self_esteem', 'locus_of_control', 'cog_flexibility']:
 88            value = getattr(self, attr)
 89            if not 0 <= value <= 1:
 90                raise ValueError(f"{attr} must be between 0 and 1, got {value}")
 91    
 92    def to_dict(self) -> Dict[str, float]:
 93        return asdict(self)
 94

104@dataclass
105class SocialPersonalityModel:
106    """Defines NPC's social behavior and ideological stance"""
107    assertion: float = 0.5  # 0=passive/easily persuaded, 1=assertive/challenges player
108    conf_indep: float = 0.5  # 0=conformist (follows group), 1=independent (autonomous)
109    empathy: float = 0.5  # 0=self-focused, 1=empathetic
110    
111    # Ideology as key-value pairs (weights sum to 1.0)
112    ideology: Dict[str, float] = field(default_factory=dict)
113    
114    # Psychological modifier (e.g., "Martyr", "Napoleon Complex", "Inferiority Complex")
115    wildcard: Optional[str] = None
116    
117    # Faction affiliation
118    faction: Optional[str] = None
119    social_position: SocialPosition = SocialPosition.MEMBER
120    
121    def __post_init__(self):
122        self._validate_ranges()
123        self._normalize_ideology()
124    
125    def _validate_ranges(self):
126        """Ensure float values are within 0-1 range"""
127        for attr in ['assertion', 'conf_indep', 'empathy']:
128            value = getattr(self, attr)
129            if not 0 <= value <= 1:
130                raise ValueError(f"{attr} must be between 0 and 1, got {value}")
131    
132    def _normalize_ideology(self):
133        """Ensure ideology weights sum to 1.0"""
134        if self.ideology:
135            total = sum(self.ideology.values())
136            if total > 0:
137                self.ideology = {k: v/total for k, v in self.ideology.items()}
138    
139    def get_dominant_ideology(self) -> Optional[str]:
140        """Returns the ideology with highest weight"""
141        if not self.ideology:
142            return None
143        return max(self.ideology, key=self.ideology.get)
144    
145    def get_wildcard_config(self) -> Dict[str, Any]:
146        """
147        Look up this NPC's wildcard in the global wildcard→Ollama-config index.
148
149        Returns a (possibly empty) dict which may contain:
150        - absolute overrides: temperature, top_p, top_k, repeat_penalty, num_predict, stop
151        - relative modifiers: temp_offset (float added to base temperature)
152        """
153        if not self.wildcard:
154            return {}
155        return WILDCARD_OLLAMA_CONFIG.get(self.wildcard, {})
156    
157    def to_dict(self) -> Dict[str, Any]:
158        data = asdict(self)
159        data['social_position'] = self.social_position.value
160        return data
161    
162    @classmethod
163    def from_dict(cls, data: Dict[str, Any]) -> 'SocialPersonalityModel':
164        if 'social_position' in data:
165            data['social_position'] = SocialPosition(data['social_position'])
166        return cls(**data)

173@dataclass
174class WorldPerceptionModel:
175    """Stores NPC's knowledge and relationship with player"""
176    world_history: str = ""
177    personal_history: str = ""
178    player_history: str = ""
179    player_relation: float = 0.5  # 0=contempt, 0.5=neutral, 1=friendship
180    
181    def __post_init__(self):
182        if not 0 <= self.player_relation <= 1:
183            raise ValueError(f"player_relation must be between 0 and 1, got {self.player_relation}")
184    
185    def update_relation(self, delta: float):
186        """Adjust relation value, clamped to 0-1"""
187        self.player_relation = max(0.0, min(1.0, self.player_relation + delta))
188    
189    def update_personal_history(self, event: str):
190        """Append to personal history"""
191        separator = "\n- " if self.personal_history else "- "
192        self.personal_history += separator + event
193    
194    def update_player_history(self, event: str):
195        """Append to player history"""
196        separator = "\n- " if self.player_history else "- "
197        self.player_history += separator + event
198    
199    def to_dict(self) -> Dict[str, Any]:
200        return asdict(self)
201    
202    @classmethod
203    def from_dict(cls, data: Dict[str, Any]) -> 'WorldPerceptionModel':
204        """Create WorldPerceptionModel from dictionary"""
205        return cls(
206            world_history=data.get('world_history', ""),
207            personal_history=data.get('personal_history', ""),
208            player_history=data.get('player_history', ""),
209            player_relation=data.get('player_relation', 0.5)
210        )

217@dataclass
218class NPCModel:
219    """Complete NPC personality and state"""
220    name: str
221    age: int
222    cognitive: CognitiveModel
223    social: SocialPersonalityModel
224    world: WorldPerceptionModel
225    
226    # Temporary modifiers (reset after conversation)
227    _temp_modifiers: Dict[str, float] = field(default_factory=dict, repr=False)
228    
229    def get_attribute(self, attr_path: str) -> Any:
230        """
231        Retrieve attribute using dot notation
232        Example: 'cognitive.self_esteem' or 'social.assertion'
233        """
234        parts = attr_path.split('.')
235        obj = self
236        for part in parts:
237            obj = getattr(obj, part)
238        return obj
239    
240    def apply_temp_mod(self, attr_path: str, modifier: float):
241        """Apply temporary modifier to an attribute (conversation-scoped)"""
242        original = self.get_attribute(attr_path)
243        if attr_path not in self._temp_modifiers:
244            self._temp_modifiers[attr_path] = original
245        
246        # Apply modifier
247        parts = attr_path.split('.')
248        obj = self
249        for part in parts[:-1]:
250            obj = getattr(obj, part)
251        
252        final_attr = parts[-1]
253        new_value = max(0.0, min(1.0, modifier))  # Clamp to 0-1
254        setattr(obj, final_attr, new_value)
255    
256    def reset_temp_mods(self):
257        """Reset all temporary modifiers after conversation"""
258        for attr_path, original_value in self._temp_modifiers.items():
259            parts = attr_path.split('.')
260            obj = self
261            for part in parts[:-1]:
262                obj = getattr(obj, part)
263            setattr(obj, parts[-1], original_value)
264        self._temp_modifiers.clear()
265    
266    def to_dict(self) -> Dict[str, Any]:
267        """Serialize to dictionary"""
268        return {
269            'name': self.name,
270            'age': self.age,
271            'cognitive': self.cognitive.to_dict(),
272            'social': self.social.to_dict(),
273            'world': self.world.to_dict()
274        }
275    
276    def to_json(self, filepath: Optional[str] = None) -> str:
277        """Serialize to JSON string or file"""
278        data = self.to_dict()
279        json_str = json.dumps(data, indent=2)
280        
281        if filepath:
282            with open(filepath, 'w') as f:
283                f.write(json_str)
284        
285        return json_str
286    
287    @classmethod
288    def from_dict(cls, data: Dict[str, Any]) -> 'NPCModel':
289        """Deserialize from dictionary"""
290        return cls(
291            name=data['name'],
292            age=data.get('age', 30),
293            cognitive=CognitiveModel.from_dict(data['cognitive']),
294            social=SocialPersonalityModel.from_dict(data['social']),
295            world=WorldPerceptionModel.from_dict(data['world'])
296        )
297    
298    @classmethod
299    def from_json(cls, json_source: str) -> 'NPCModel':
300        """
301        Deserialize from JSON string or file path
302        Automatically detects if input is filepath or JSON string
303        """
304        try:
305            # Try as filepath first
306            with open(json_source, 'r') as f:
307                data = json.load(f)
308        except (FileNotFoundError, OSError):
309            # Parse as JSON string
310            data = json.loads(json_source)
311        
312        return cls.from_dict(data)
